panama_survey_decision = {
	is_shown = {
		has_interest_marker_in_region = region_central_america
		NOT = { 
			has_variable = survey_panama_taken 
			has_global_variable = panama_canal_complete
		}
		OR = {
			country_rank = rank_value:great_power
			s:STATE_BATAVIA = {
				any_scope_state = {
					owner = ROOT
				}
			}
		}
		s:STATE_BATAVIA = {
			any_scope_state = {
				NOT = {
					any_scope_building = { 
						is_building_type = building_panama_canal 
					}
				}
			}
		}
	}

	possible = {
		has_technology_researched = civilizing_mission
		produced_bureaucracy > 1000
	}
	
	when_taken = {
		set_variable = {
			name = survey_panama_taken
			value = 0
		}
		add_modifier = {
			name = modifier_surveying_panama
		}
		add_journal_entry = {
			type = je_panama_survey
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = { produced_bureaucracy > 5000 }
			add = 5
		}
		modifier = {
			trigger = {
				has_strategy = ai_strategy_armed_isolationism
			}
			add = -1000
		}		
		modifier = {
			trigger = {
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}
			add = -1000
		}		
	}
}

stop_panama_survey_decision = {
	is_shown = {
		has_modifier = modifier_surveying_panama
	}
	possible = {
		has_modifier = modifier_surveying_panama
	}
	when_taken = {
		remove_variable = survey_panama_taken
		remove_modifier = modifier_surveying_panama
	}
	ai_chance = {
		base = 0
	}
}

panama_canal_purchase = {
	is_shown = {
		is_subject = no
		has_variable = panama_canal_purchase
		NOT = { 
			has_global_variable = panama_canal_purchase_var
			has_global_variable = panama_canal_complete
		}
		s:STATE_BATAVIA = {
			any_scope_state = {
				NOT = {
					any_scope_building = { 
						is_building_type = building_panama_canal 
					}
				}
				owner = {
					owns_treaty_port_in = STATE_BATAVIA
					NOT = { 
						has_variable = panama_survey_complete 
					}
				}
			}
		}
	}

	possible = {
		gdp > 50000000
		p:xD0EF40.state.owner = {
			relations:root >= relations_threshold:cordial
		} 
		NOT = { has_war_with = ROOT }
	}

	when_taken = {
		set_global_variable = panama_canal_purchase_var
		if = {
			limit = {
				s:STATE_BATAVIA = {
					any_scope_state = {
						owner = { 
							owns_treaty_port_in = STATE_BATAVIA
							is_player = no 
						}
					}
				}
			}
			hidden_effect = {
				s:STATE_BATAVIA = {
					random_scope_state = {
						limit = {
							owner = {
								owns_treaty_port_in = STATE_BATAVIA
							}
						}
						owner = {
							add_modifier = {
								name = panama_sale
								months = 120
							}
						}
					}
				}
			}
			s:STATE_BATAVIA = {
				set_owner_of_provinces = {
					country = ROOT
					provinces = {xD0EF40}
				}
			}
			add_modifier = {
				name = panama_purchase
				months = 120
			}
		}
		else = {
			trigger_event = {
				id = canal_events.5
				days = 0
				popup = yes
			}
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = { gold_reserves > 15000000 }
			add = 10
		}
		modifier = {
			trigger = { 
				weekly_net_fixed_income > 100000
				gold_reserves > 0
			}
			add = 10
		}
		modifier = {
			trigger = {
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}
			add = -1000
		}		
	}
}
