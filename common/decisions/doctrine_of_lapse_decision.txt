doctrine_of_lapse = {
	is_shown = {
		c:PER ?= this
		any_subject_or_below = {
			is_subject_type = subject_type_satrapy
		}
		NOT = { has_variable = doctrine_of_lapse_ended }
	}

	possible = {
		is_at_war = no
		custom_tooltip = {
			text = doctrine_of_lapse_cooldown_tt
			NOT = { has_variable = doctrine_of_lapse_cooldown }
		}
		THIS = c:PER
		any_subject_or_below = {
			is_player = no
			country_is_in_india = yes
			is_subject_type = subject_type_satrapy
			OR = {
				ruler = {
					OR = {
						has_trait = senile
						has_trait = psychological_affliction
						has_trait = sickly
						has_trait = cancer
						has_trait = tuberculosis
						has_trait = opium_addiction
						has_trait = cocaine_addiction
						has_trait = grifter
					}
				}
				AND = {
					ruler = {
						age > 30
					}
					NOT = {
						any_scope_character = {
							is_heir = yes
						}
					}
				}
				any_scope_character = {
					is_heir = yes
					is_female = yes
				}
				trigger_if = {
					limit = {
						c:BUN ?= prev
					}
					custom_tooltip = {
						text = damodar_rao_tt
						has_variable = illegitimate_can_annex
					}
				}
			}
		}
	}

	when_taken = {
		trigger_event = { id = raj.2 days = 0 popup = yes }
		set_variable = {
			name = doctrine_of_lapse_cooldown
			years = 2
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				infamy < infamy_threshold:infamous
			}
			add = 1000
		}
	}
}
