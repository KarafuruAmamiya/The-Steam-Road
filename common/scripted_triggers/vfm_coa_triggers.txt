
coa_def_protectorate_trigger = {
	OR = {
		NOT = { exists = scope:target }
		AND = {
			exists = scope:target
			scope:target = { 
				OR = {
					is_subject_type = subject_type_protectorate
				}
			}
		}
	}
}

coa_def_luso_american_trigger = {
    any_primary_culture = {
		OR = {
			has_discrimination_trait = lusophone
			has_discrimination_trait = hispanophone
		}
    	has_discrimination_trait = south_european_heritage
	    OR = {
			has_discrimination_trait = american
			has_discrimination_trait = laurentian
		}
    }
}
