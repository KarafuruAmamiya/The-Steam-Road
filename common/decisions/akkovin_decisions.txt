proclaim_erzurum_irade = {

	is_shown = {
		exists = c:AKN
		root = c:AKN
		NOT = { has_variable = erzurum_edict }
	}

	possible = {
		custom_tooltip = {
			text = erzurum_edict_tt
			has_modifier = treaty_of_baglar
		}
	}

	when_taken = {
		set_variable = erzurum_edict
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:sunni 
				}
			}
			add_radicals = {
				religion = rel:sunni
				value = medium_radicals
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:orthodox 
				}
			}
			add_radicals = {
				religion = rel:orthodox
				value = medium_radicals
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:oriental_orthodox 
				}
			}
			add_radicals = {
				religion = rel:oriental_orthodox
				value = medium_radicals
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:catholic 
				}
			}
			add_radicals = {
				religion = rel:catholic
				value = medium_radicals
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:protestant 
				}
			}
			add_radicals = {
				religion = rel:protestant
				value = medium_radicals
			}
		}
		if = {
			limit = {
				any_scope_pop = { 
					religion = rel:jewish 
				}
			}
			add_radicals = {
				religion = rel:jewish
				value = medium_radicals
			}
		}
	}

	ai_chance = {
		base = 10
	}
}

centralise_kurdish_territories = {
	is_shown = {
		exists = c:AKN
		root = c:AKN
		has_journal_entry = je_akkovin_reform_kurdish
	}

	possible = {
		is_at_war = no
		custom_tooltip = {
			text = centralise_kurdish_territories_tt
			NOT = { has_variable = centralise_kurdish_vilayets_cooldown }
		}
		any_scope_state = {
			has_modifier = kurdish_autonomy
		}
	}

	when_taken = {
		trigger_event = { id = akkovin_flavor.3 days = 0 popup = yes }
		set_variable = {
			name = centralise_kurdish_vilayets_cooldown
			years = 2
		}
	}

	ai_chance = {
		base = 10
	}
}
