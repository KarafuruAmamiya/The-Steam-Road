je_opium_obsession = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = opium_wars.6
			5 = opium_wars.7
			5 = opium_wars.8
		}
		effect = {
			if = {
				limit = {
					is_banning_goods = opium
					is_at_war = no
					NOT = {
						any_scope_building = {
							is_building_type = building_illegal_opium_plantation
						}
					}
				}
				change_variable = {
					name = opium_wars_completion_tracker
					add = 1
				}
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		custom_tooltip = {
			text = opium_ban_tt
			is_banning_goods = opium
		}
		is_at_war = no
		NOT = {
			any_scope_building = {
				is_building_type = building_illegal_opium_plantation
			}
		}
	}

	on_complete = {
		if = {
			limit = { has_variable = opium_wars_aggressor }
			remove_variable = opium_wars_aggressor
		}
		trigger_event = {
			id = opium_wars.3
		}
	}

	current_value = {
		value = root.var:opium_wars_completion_tracker
	}

	goal_add_value = {
		add = 60
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}

je_opium_wars = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	complete = {
		OR = {
			NOT = { exists = c:PER }
			AND = {
				exists = c:PER
				any_scope_state = {
					is_treaty_port = yes
					state_region = {
						any_scope_state = {
							owner = c:PER
						}
					}
				}
				c:PER = {
					is_banning_goods = opium
				}
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = opium_wars.4
			days = 5
		}
		hidden_effect = {
			c:PER = {
				trigger_event = {
					id = opium_wars.4
					days = 5
				}
			}
		}
	}

	fail = {
		custom_tooltip = {
			text = opium_wars_fail_tt
			AND = {
				exists = c:PER
				has_diplomatic_pact = {
					who = c:PER
					type = war_reparations
				}
			}
		}
	}

	on_fail = {
		custom_tooltip = {
			text = opium_wars_fail_invalid_tt
			remove_variable = opium_wars_aggressor
			any_country = {
				limit = {
					has_variable = opium_wars_exporter
				}
				remove_variable = opium_wars_exporter
			}
		}
	}

	on_timeout = {
		custom_tooltip = {
			text = opium_wars_fail_invalid_tt
			remove_variable = opium_wars_aggressor
			any_country = {
				limit = {
					has_variable = opium_wars_exporter
				}
				remove_variable = opium_wars_exporter
			}
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes

	timeout = 1825
}
