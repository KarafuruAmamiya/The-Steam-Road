vfm_released_independent_recognized = {
	effect = {
		if = {
			limit = {
				OR = {
					is_country_type = recognized
					is_country_type = colonial
				}
				country_rank >= rank_value:great_power
				scope:target = {
					is_country_type = unrecognized
				}
			}
			scope:target = {
				set_country_type = recognized
			}
		}
	}
}

vfm_barracks_colony_created = {
	effect = {
		#temporary solution giving barracks to colonies to prevent being overrun by uprisings instantly
		if = {
			limit = {
				owner = {
					country_is_in_africa = no
					is_ai = yes
				}
				is_under_colonization = yes
				is_incorporated = no
				OR = {
					state_in_africa = yes
				}
				NOT = {
					has_building = building_barracks
				}
			}
			create_building = {
				building = building_barracks
				level = 1
			}
		}
	}
}

vfm_vice_presidents = {
	effect = {
		if = { #Creates initial vice president if conditions met
			limit = {
				has_law = law_type:law_presidential_republic
				country_has_voting_franchise = yes
				NOT = { has_variable = vice_president }
			}
			create_character = {
				on_created = {
					set_variable = vice_president_trigger
					owner = {
						set_variable = {
							name = vice_president
							value = PREV
						}
					}
				}		
			}
		}
		if = { #Removes vice president if no longer presidential republic
			limit = {
				has_law = law_type:law_presidential_republic
				country_has_voting_franchise = yes
				has_variable = vice_president
				NOT = { 
					any_scope_character = {
						has_variable = vice_president_trigger
					}
				}
			}
			#if = {
			#	limit = {
			#		ruler = {
			#			has_variable = vice_president_trigger
			#		}
			#	}
			#	ruler = { remove_variable = vice_president_trigger }
			#}
			if = {
				limit = {
					any_scope_character = {
						is_ruler = no
						is_heir = no
						NOT = {
							OR = {
								has_role = general
								has_role = admiral
								has_role = agitator
							}
						}
						interest_group = {
							is_in_government = yes
						}
					}
				}
				random_scope_character = {
					limit = {
						is_ruler = no
						is_heir = no
						NOT = {
							OR = {
								has_role = general
								has_role = admiral
								has_role = agitator
							}
						}
						interest_group = {
							is_in_government = yes
						}
					}
					set_variable = vice_president_trigger
					save_scope_as = new_vice_president
					owner = {
						set_variable = {
							name = vice_president
							value = PREV
						}
					}
				}
			}
			else = {
				create_character = {
					on_created = {
						set_variable = vice_president_trigger
						save_scope_as = new_vice_president
						owner = {
							set_variable = {
								name = vice_president
								value = PREV
							}
						}
					}		
				}
			}
		}	
		if = { #Removes vice president if no longer presidential republic
			limit = {
				has_variable = vice_president
				NOT = { has_law = law_type:law_presidential_republic }
			}
			remove_variable = vice_president
			random_scope_character = {
				limit = {
					has_variable = vice_president_trigger
				}
				remove_variable = vice_president_trigger
			}
		}
	}
}

vfm_prime_ministers = {
	effect = {
		if = {
			limit = {
				OR = {
					has_law = law_type:law_monarchy
					has_law = law_type:law_parliamentary_republic
				}
				country_has_voting_franchise = yes
				NOT = { has_variable = prime_minister }
			}
			random_scope_character = {
				limit = {
					is_ruler = no
					is_heir = no
					NOT = {
						OR = {
							has_role = general
							has_role = admiral
							has_role = agitator
						}
					}
					interest_group = {
						is_in_government = yes
					}
				}
				set_variable = prime_minister_trigger
				owner = {
					set_variable = {
						name = prime_minister
						value = PREV
					}
				}
			}
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_monarchy
					has_law = law_type:law_parliamentary_republic
				}
				country_has_voting_franchise = yes
				has_variable = prime_minister
				NOT = { 
					any_scope_character = {
						has_variable = prime_minister_trigger
					}
				}

				#any_scope_character = {
				#	AND = {
				#		NOT = { ig = is_in_government = yes }
				#		has_variable = prime_minister_trigger
				#	}
				#}
			}
			random_scope_character = {
				limit = {
					is_ruler = no
					is_heir = no
					NOT = {
						OR = {
							has_role = general
							has_role = admiral
							has_role = agitator
						}
					}
					interest_group = {
						is_in_government = yes
					}
				}
				set_variable = prime_minister_trigger
				save_scope_as = new_prime_minister
				owner = {
					set_variable = {
						name = prime_minister
						value = PREV
					}
				}
			}
			every_scope_character = {
				limit = {
					has_variable = prime_minister_trigger
					NOT = { THIS = scope:new_prime_minister }
				}
				remove_variable = prime_minister_trigger
			}
		}
		if = {
			limit = {
				NOT = {
					AND = {
						country_has_voting_franchise = yes
						OR = {
							has_law = law_type:law_monarchy
							has_law = law_type:law_parliamentary_republic
						}
					}
				}
				has_variable = prime_minister
			}
			remove_variable = prime_minister
			random_scope_character = {
				limit = {
					has_variable = prime_minister_trigger
				}
				remove_variable = prime_minister_trigger
			}
		}
	}
}
