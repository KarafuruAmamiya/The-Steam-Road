# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law
# cultural_acceptance_rule and religious_acceptance_rule are triggers called in culture scope with scope:country as the country with the laws

law_ethnostate = {
	group = lawgroup_citizenship
	
	icon = "gfx/interface/icons/law_icons/ethnostate.dds"
	
	progressiveness = -100
	
	on_enact = {
		recalculate_pop_ig_support = yes
	}
	
	cultural_acceptance_rule = {
		is_primary_culture_of = scope:country
	}

	unlocking_technologies = {
		nationalism
	}
	
	modifier = {
		country_authority_add = 200
		state_radicals_from_sol_change_accepted_culture_mult = -0.2
		state_loyalists_from_sol_change_accepted_culture_mult = 0.2			
	}
	
	possible_political_movements = {
		law_racial_segregation
		law_cultural_exclusion
	}

	pop_support = {
		value = 0
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}

law_shared_heritage = {
	group = lawgroup_citizenship
	
	icon = "gfx/interface/icons/law_icons/racial_segregation.dds"
	
	progressiveness = 25
	
	can_enact = {
		any_primary_culture = {
			OR = {
				has_discrimination_trait = eastern_european_heritage
				has_discrimination_trait = western_european_heritage
				has_discrimination_trait = south_european_heritage
				has_discrimination_trait = anglophone
				has_discrimination_trait = francophone
				has_discrimination_trait = hispanophone
				has_discrimination_trait = lusophone
				has_discrimination_trait = scandinavian_culture_group
				has_discrimination_trait = beneluxian_culture_group
				has_discrimination_trait = turkic_heritage
				has_discrimination_trait = iranian_heritage
				has_discrimination_trait = arabic_heritage
				has_discrimination_trait = east_asian_heritage
				has_discrimination_trait = north_asian_heritage
				has_discrimination_trait = mongol_heritage
				has_discrimination_trait = southeast_asian_heritage
				has_discrimination_trait = malay_heritage
			}
		}
	}
	
	is_visible = {
		any_primary_culture = {
			OR = {
				has_discrimination_trait = eastern_european_heritage
				has_discrimination_trait = western_european_heritage
				has_discrimination_trait = south_european_heritage
				has_discrimination_trait = anglophone
				has_discrimination_trait = francophone
				has_discrimination_trait = hispanophone
				has_discrimination_trait = lusophone
				has_discrimination_trait = scandinavian_culture_group
				has_discrimination_trait = beneluxian_culture_group
				has_discrimination_trait = turkic_heritage
				has_discrimination_trait = iranian_heritage
				has_discrimination_trait = arabic_heritage
				has_discrimination_trait = east_asian_heritage
				has_discrimination_trait = north_asian_heritage
				has_discrimination_trait = mongol_heritage
				has_discrimination_trait = southeast_asian_heritage
				has_discrimination_trait = malay_heritage
			}
		}
	}
	
	on_enact = {
		recalculate_pop_ig_support = yes
	}

	cultural_acceptance_rule = {
		custom_tooltip = {
			text = shared_heritage_tt
			trigger_if = {
				limit = {
					scope:country = {
						any_primary_culture = {
							OR = {
								has_discrimination_trait = eastern_european_heritage
								has_discrimination_trait = western_european_heritage
								has_discrimination_trait = south_european_heritage
								has_discrimination_trait = anglophone
								has_discrimination_trait = francophone
								has_discrimination_trait = hispanophone
								has_discrimination_trait = lusophone
								has_discrimination_trait = scandinavian_culture_group
								has_discrimination_trait = beneluxian_culture_group
							}
						}
					}
				}
				OR = {
					has_discrimination_trait = eastern_european_heritage
					has_discrimination_trait = western_european_heritage
					has_discrimination_trait = south_european_heritage
				}
			}
			trigger_else_if = {
				limit = {
					scope:country = {
						any_primary_culture = {
							OR = {
								has_discrimination_trait = turkic_heritage
								has_discrimination_trait = iranian_heritage
								has_discrimination_trait = arabic_heritage
							}
						}
					}
				}
				OR = {
					has_discrimination_trait = iranian_heritage
					has_discrimination_trait = turkic_heritage
					has_discrimination_trait = arabic_heritage
				}
			}
			trigger_else_if = {
				limit = {
					scope:country = {
						any_primary_culture = {
							OR = {
								has_discrimination_trait = east_asian_heritage
								has_discrimination_trait = north_asian_heritage
								has_discrimination_trait = mongol_heritage
							}
						}
					}
				}
				OR = {
					has_discrimination_trait = east_asian_heritage
					has_discrimination_trait = north_asian_heritage
					has_discrimination_trait = mongol_heritage
				}
			}
			trigger_else_if = {
				limit = {
					scope:country = {
						any_primary_culture = {
							OR = {
								has_discrimination_trait = southeast_asian_heritage
								has_discrimination_trait = malay_heritage
							}
						}
					}
				}
				OR = {
					has_discrimination_trait = southeast_asian_heritage
					has_discrimination_trait = malay_heritage
				}
			}
		}
	}

		state_loyalists_from_sol_change_accepted_culture_mult = 0.075		
	}
	
	possible_political_movements = {
		law_national_supremacy
		law_racial_segregation
		law_cultural_exclusion
	}
	
	pop_support = {
		value = 0
		
		add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					culture_accepted = no
					culture = {
						OR = {
							has_discrimination_trait = eastern_european_heritage
							has_discrimination_trait = western_european_heritage
							has_discrimination_trait = south_european_heritage
						}
					}
					owner = { has_law = law_type:law_racial_segregation }
