fra_expand_north_pacific_company = {
	is_shown = {
		country_is_in_europe = yes
		exists = c:BRL
		c:BRL = {
			is_subject_of = ROOT
			is_subject_type = subject_type_chartered_company
		}
	}
	possible = {
		OR = { 
			any_state = {
				owner = ROOT
				OR = {
					region = sr:region_east_siberia
					region = sr:region_manchuria
				}
			}
			any_subject_or_below = {
				OR = {
					capital.region = sr:region_east_siberia
					capital.region = sr:region_manchuria
				}
				NOT = { 
					THIS = c:BRL
					is_subject_of = c:BRL
				}
			}
		}
	}
	when_taken = {
		custom_tooltip = expand_north_pacific_company_tt
		if = {
			limit= {
				any_state = {
					owner = ROOT
					OR = {
						region = sr:region_east_siberia
						region = sr:region_manchuria
					}
				}
			}
			every_scope_state = {
				limit = {
					owner = ROOT
					OR = {
						region = sr:region_east_siberia
						region = sr:region_manchuria
					}
				}
				set_state_owner = c:BRL
				set_state_type = unincorporated
			}
		}
		if = {
			limit= {
				any_subject_or_below = {
					OR = {
						capital.region = sr:region_east_siberia
						capital.region = sr:region_manchuria
					}
					NOT = { 
						THIS = c:BRL 
						is_subject_of = c:BRL 
					}
				}
			}
			every_subject_or_below = {
				if = {
					limit = { 
						is_subject_type = subject_type_protectorate
						OR = {
							capital.region = sr:region_east_siberia
							capital.region = sr:region_manchuria
						}
						NOT = { 
							THIS = c:BRL 
							is_subject_of = c:BRL 
						} 						
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_brl_protectorate
					c:BRL = {
						create_diplomatic_pact = {
							country = scope:new_brl_protectorate
							type = protectorate
						}
					}
				}
				if = {
					limit = { 
						is_subject_type = subject_type_vassal
						OR = {
							capital.region = sr:region_east_siberia
							capital.region = sr:region_manchuria
						}
						NOT = { 
							THIS = c:BRL
							is_subject_of = c:BRL 
						} 
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_brl_vassal
					c:BRL = {
						create_diplomatic_pact = {
							country = scope:new_brl_vassal
							type = puppet
						}
					}
				}
				if = {
					limit = { 
						is_subject_type = subject_type_puppet
						OR = {
							capital.region = sr:region_east_siberia
							capital.region = sr:region_manchuria
						}
						NOT = { 
							THIS = c:BRL
							is_subject_of = c:BRL 
						} 
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_brl_puppet
					c:BRL = {
						create_diplomatic_pact = {
							country = scope:new_brl_puppet
							type = puppet
						}
					}
				}
			}
		}
	}
	ai_chance = {
		base = 100
	}
}
