					country_or_subject_owns_entire_state_region = STATE_MEDINA

						limit = {
							exists = c:NEJ
						}
						c:NEJ = {
							is_subject_of = ROOT
						}
					}
					trigger_else = {
						has_state_in_state_region = STATE_NEJD
					}
					trigger_if = {
						limit = {
							exists = c:JAB
						}
						c:JAB = {
							is_subject_of = ROOT
						}
					}
					trigger_else = {
						has_state_in_state_region = STATE_HAIL
					}
				}
			}
			exists = c:ARA
		}
	}
	
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					exists = c:PER

						NOT = { has_variable = pre_treaty }

							who = c:AKN
							attitude = protective
						}
					}
				}
				random_list = {
					9 = { }
					1 = {
						c:PER = { trigger_event = akkovin_crisis.4 }
					}
				}
			}
			if = {
				limit = {
					has_variable = completed_akkovin_reform_economy_je
					has_variable = completed_akkovin_reform_education_je
					has_variable = completed_akkovin_reform_bureaucracy_je
