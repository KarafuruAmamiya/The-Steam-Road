ig_desires_laws_for_petition = {
	any_interest_group = {
		is_in_government = yes
		is_marginal = no
		any_preferred_law = {
			save_temporary_scope_as = petition_law_trig
			prev = {
				law_stance = {
					law = scope:petition_law_trig.type
					value > neutral
				}
			}
			NOT = {
				ROOT = {
					has_law = prev.type
				}
			}  
			can_be_enacted = yes
		}
	}
}
