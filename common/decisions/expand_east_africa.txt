fra_expand_east_africa = {
	is_shown = {
		country_is_in_europe = yes
		exists = c:FEA
		c:FEA = {
			is_subject_of = ROOT
			is_subject_type = subject_type_dominion
		}
	}
	possible = {
		OR = { 
			any_state = {
				owner = ROOT
				OR = {
					region = sr:region_nile_basin
					region = sr:region_ethiopia
					region = sr:region_zanj
				}
				is_under_colonization = no
			}
			any_subject_or_below = {
				OR = {
					capital.region = sr:region_nile_basin
					capital.region = sr:region_ethiopia
					capital.region = sr:region_zanj
				}
				NOT = { 
					THIS = c:FEA
					is_subject_of = c:FEA
				}
			}
		}
	}
	when_taken = {
		custom_tooltip = expand_east_africa_tt
		if = {
			limit= {
				any_state = {
					owner = ROOT
					OR = {
						region = sr:region_nile_basin
						region = sr:region_ethiopia
						region = sr:region_zanj
					}
				}
			}
			every_scope_state = {
				limit = {
					owner = ROOT
					OR = {
						region = sr:region_nile_basin
						region = sr:region_ethiopia
						region = sr:region_zanj
					}
				}
				set_state_owner = c:FEA
				set_state_type = unincorporated
			}
		}
		if = {
			limit= {
				any_subject_or_below = {
					OR = {
						capital.region = sr:region_nile_basin
						capital.region = sr:region_ethiopia
						capital.region = sr:region_zanj
					}
					NOT = { 
						THIS = c:FEA 
						is_subject_of = c:FEA 
					}
				}
			}
			every_subject_or_below = {
				if = {
					limit = { 
						is_subject_type = subject_type_protectorate
						OR = {
							capital.region = sr:region_nile_basin
							capital.region = sr:region_ethiopia
							capital.region = sr:region_zanj
						}
						NOT = { 
							THIS = c:FEA 
							is_subject_of = c:FEA 
						} 						
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_fea_protectorate
					c:FEA = {
						create_diplomatic_pact = {
							country = scope:new_fea_protectorate
							type = protectorate
						}
					}
				}
				if = {
					limit = { 
						is_subject_type = subject_type_vassal
						OR = {
							capital.region = sr:region_nile_basin
							capital.region = sr:region_ethiopia
							capital.region = sr:region_zanj
						}
						NOT = { 
							THIS = c:FEA
							is_subject_of = c:FEA 
						} 
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_fea_vassal
					c:FEA = {
						create_diplomatic_pact = {
							country = scope:new_fea_vassal
							type = puppet
						}
					}
				}
				if = {
					limit = { 
						is_subject_type = subject_type_puppet
						OR = {
							capital.region = sr:region_nile_basin
							capital.region = sr:region_ethiopia
							capital.region = sr:region_zanj
						}
						NOT = { 
							THIS = c:FEA
							is_subject_of = c:FEA 
						} 
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_fea_puppet
					c:FEA = {
						create_diplomatic_pact = {
							country = scope:new_fea_puppet
							type = puppet
						}
					}
				}
			}
		}
	}
	ai_chance = {
		base = 100
	}
}
