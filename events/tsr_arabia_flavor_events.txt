# Spread of Nahḍa (Arabic Renaissance)
arabia_flavor.1 = {
	type = country_event
	placement = scope:nahda_state
	title = arabia_flavor.1.t
	desc = arabia_flavor.1.d
	flavor = arabia_flavor.1.f

	duration = 3

	event_image = {
		video = "middleeast_islam"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_academia.dds"

	trigger = {
		game_date >= 1845.1.1
		has_technology_researched = romanticism
		#Link to Akkovin reforms
		NOT = { has_global_variable = nahda_start }
		any_scope_state = {
			OR = {
				state_is_in_middle_east = yes
				state_is_in_north_africa = yes
			}
			OR = {
				state_region = { 
					is_homeland = cu:levantine 
				}
				state_region = { 
					is_homeland = cu:mashriqi 
				}
				state_region = { 
					is_homeland = cu:iraqi 
				}
				state_region = { 
					is_homeland = cu:misri 
				}
				state_region = { 
					is_homeland = cu:maghrebi 
				}
			}
			state_population >= 300000
			total_urbanization >= 1
		}
	}

	immediate = {
		set_global_variable = nahda_start
		random_scope_state = {
			limit = {
				OR = {
					state_is_in_middle_east = yes
					state_is_in_north_africa = yes
				}
				OR = {
					state_region = { 
						is_homeland = cu:levantine 
					}
					state_region = { 
						is_homeland = cu:mashriqi 
					}
					state_region = { 
						is_homeland = cu:iraqi 
					}
					state_region = { 
						is_homeland = cu:misri 
					}
					state_region = { 
						is_homeland = cu:maghrebi 
					}
				}
				state_population >= 300000 
			}
			save_scope_as = nahda_state
		}
		if = {
			limit = {
				any_primary_culture = {
					has_discrimination_trait = arabic_heritage
				}
				OR = {
					country_is_in_middle_east = yes
					AND = { exists = c:TUN c:TUN = this }
					AND = { exists = c:MOR c:MOR = this }
					AND = { exists = c:EGY c:EGY = this }
				}
				NOT = { has_technology_researched = nationalism }
			}
			add_technology_progress = { progress = 2500 technology = nationalism }
		}
		every_scope_culture = {
			limit = {
				has_discrimination_trait = arabic_heritage
				NOT = { has_discrimination_trait = jewish_group }
			}
			save_scope_as = arab_culture
		}
	}

	option = { #Encourage the movement
		name = arabia_flavor.1.a
		default_option = yes
		scope:nahda_state = {
			add_modifier = {
				name = nahda_center
				months = -1
			}
		}
		ig:ig_intelligentsia = {
			add_modifier = {
				name = nahda_ig_modifier
				months = -1
			}
		}
		ai_chance = {
			base = 100
			modifier = {
				NOT = {
					any_primary_culture = {
						has_discrimination_trait = arabic_heritage
					}
				}
				add = -1000
			}		
		}
	}
	
	option = { #Suppress the movement
		name = arabia_flavor.1.b
		scope:nahda_state = {
			add_modifier = {
				name = nahda_stronghold
				months = -1
			}
		}
		add_radicals = {
			value = large_radicals
			culture = scope:arab_culture
        }
		ai_chance = {
			base = 0
			modifier = {
				NOT = {
					any_primary_culture = {
						has_discrimination_trait = arabic_heritage
					}
				}
				add = 1000
			}
		}
	}
}

# Butrus Al-Bustani
arabia_flavor.2 = {
	type = country_event
	placement = scope:lebanon_scope
	title = arabia_flavor.2.t
	desc = arabia_flavor.2.d
	flavor = arabia_flavor.2.f

	duration = 3

	event_image = {
		video = "middleeast_engineer_blueprint"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	minor_left_icon = scope:al_bustani_scope

	icon = "gfx/interface/icons/event_icons/event_academia.dds"

	trigger = {
		has_global_variable = nahda_start
		empty_agitator_slots >= 1
		ig:ig_intelligentsia = {
			is_marginal = no
			ig_approval <= 0	
		}
		game_date >= 1850.1.1
		game_date < 1880.1.1
		owns_entire_state_region = STATE_LEBANON
		NOT = { has_global_variable = al_bustani_var }
	}

	immediate = {
		every_overlord_or_above = {
			save_scope_as = overlord_scope
		}
		set_global_variable = al_bustani_var
		create_character = {
			template = syr_bustani_character_template
			on_created = {
				set_variable = al_bustani
			}
		}
		random_scope_character = {
			limit = {
				has_variable = al_bustani
			}
			save_scope_as = al_bustani_scope
		}
		random_country = {
			limit = {
				any_scope_state = {
					OR = {
						has_modifier = nahda_center
						has_modifier = nahda_stronghold
					}
				}
			}
			save_scope_as = nahda_country
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_LEBANON
			}
			save_scope_as = lebanon_scope
		}
	}

	option = {
		name = arabia_flavor.2.a #Suppress him
		default_option = yes
		show_as_tooltip = {
			create_character = {
				template = syr_bustani_character_template
			}
		}
		scope:al_bustani_scope = {
			add_modifier = {
				name = modifier_al_bustani_negative_popularity
				months = normal_modifier_time
			}
		}
		
		ai_chance = {
			base = 25
		}
	}
	option = {
		name = arabia_flavor.2.b #Support his independence rhetorics
		trigger = {
			country_has_primary_culture = cu:levantine
		}
		show_as_tooltip = {
			create_character = {
				template = syr_bustani_character_template
			}
		}
		every_state = {
			limit = {
				NOT = {
					owner = { country_has_primary_culture = cu:levantine }
				}
				state_region = {
					is_homeland = cu:levantine
				}
			}
			add_radicals_in_state = {
				value = medium_radicals
				culture = cu:levantine
			}
		}
		add_technology_progress = { progress = 2500 technology = nationalism }
		change_relations = {
			country = scope:overlord_scope
			value = -20
		}
		set_variable = al_bustani_nationalism
		
		ai_chance = {
			base = 0
			if = {
				limit = {
					country_has_primary_culture = cu:levantine
				}
				add = 100
			}
		}
	}
	option = {
		name = arabia_flavor.2.c #A welcome addition to our country
		show_as_tooltip = {
			create_character = {
				template = syr_bustani_character_template
			}
		}
		scope:al_bustani_scope = {
			add_modifier = {
				name = modifier_al_bustani_popularity
				months = normal_modifier_time
			}
		}
		if = {
			limit = {
				NOT = { country_has_primary_culture = cu:levantine }
			}
			every_scope_state = {
				limit = {
					state_region = {
						is_homeland = cu:levantine
					}
				}
				add_radicals_in_state = {
					value = medium_radicals
					culture = cu:levantine
				}
			}
		}
		
		ai_chance = {
			base = 25
			if = {
				limit = {
					cu:levantine = {
						culture_is_discriminated_in = root
					}
				}
				add = -100
			}
		}
	}
}
