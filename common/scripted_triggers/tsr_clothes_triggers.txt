tsr_persia_leader_trigger = {
	exists = scope:character
	scope:character = {
		exists = c:PER
		exists = home_country
		home_country = {
			OR = {
				OR = {
					country_has_primary_culture = cu:persian
					country_has_primary_culture = cu:jonubi
				}
				AND = {
					is_subject_of = c:PER
					OR = {
						exists = c:SRV c:SRV = this
						exists = c:ARD c:ARD = this
					}
				}
			}
		}
	}
}

tsr_persia_pop_trigger = {
	exists = scope:character
	scope:pop = {
		exists = c:PER
		home_country = {
			OR = {
				OR = {
					country_has_primary_culture = cu:persian
					country_has_primary_culture = cu:jonubi
				}
				AND = {
					is_subject_of = c:PER
					OR = {
						AND = { exists = c:SRV c:SRV = this }
						AND = { exists = c:TSN c:TSN = this }
						AND = { exists = c:SIN c:SIN = this }
						AND = { exists = c:ARD c:ARD = this }
					}
				}
			}
		}
	}
}

tsr_ottoman_leader_trigger = {
	exists = scope:character
	scope:character = {
		exists = home_country
		exists = c:TUR
		home_country = {
			country_has_primary_culture = cu:turkish
		}
	}
}

british_empire_clothes_trigger = {
	exists = scope:character
	scope:character = {
		OR = {
			AND = {
				exists = c:GBR
				exists = home_country
				home_country = {
					OR = {
						AND = {
							c:GBR = this
							is_subject = no # preventing uniform conflicts if GBR becomes a subject of another country with puppet uniforms
						}
						is_subject_of = c:GBR
					}		
				}
			}
			AND = {
				exists = c:ENG
				exists = home_country
				home_country = {
					OR = {
						AND = {
							c:ENG = this
							is_subject = no # preventing uniform conflicts if GBR becomes a subject of another country with puppet uniforms
						}
						is_subject_of = c:ENG
					}		
				}
			}
		}
	}
}
