ban_opium = {
	is_shown = {
		exists = market
		market.owner = root
		NOT = { is_banning_goods = g:opium }
	}

	possible = {
		NOT = { has_law = law_type:law_free_trade }
		has_law = law_type:law_substances_banned
		exists = market	
	}

	when_taken = {
		add_banned_goods = g:opium
		add_modifier = {
			name = opium_ban_authority
		}
	}
	
    ai_chance = {
        base = 0
		modifier = {
			trigger = {
				OR = {
					any_primary_culture = {
						has_cultural_obsession = opium
					}
					AND = {
						exists = ruler
						ruler = { 
							has_ideology = ideology:ideology_theocrat 
						}
					}
				}
				OR = {
					has_strategy = ai_strategy_conservative_agenda
					has_strategy = ai_strategy_reactionary_agenda
					has_strategy = ai_strategy_maintain_mandate_of_heaven
				}
			}
			add = 100
		}	 
    }	
}

			trigger = {
