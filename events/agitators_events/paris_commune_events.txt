					set_owner_of_provinces = {

					}

					set_interest_group_name = ig_national_guard
					remove_ideology = ideology_loyalist
					add_ideology = ideology_proletarian

					add_ideology = ideology_republican
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_parliamentary_republic
						}
					}
					activate_law = law_type:law_parliamentary_republic
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_elected_bureaucrats
						}
					}
					activate_law = law_type:law_elected_bureaucrats
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_universal_suffrage
						}
					}
					activate_law = law_type:law_universal_suffrage
				}

					limit = {
						NOT = {
							has_law = law_type:law_women_in_the_workplace
						}
					}
					activate_law = law_type:law_women_in_the_workplace
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_commercialized_agriculture
						}
					}
					activate_law = law_type:law_commercialized_agriculture
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_total_separation
						}
					}
					activate_law = law_type:law_total_separation
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_mass_conscription
						}
					}
					activate_law = law_type:law_mass_conscription
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_compulsory_primary_school
						}
					}
					activate_law = law_type:law_compulsory_primary_school
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_public_schools
						}
					}
					activate_law = law_type:law_public_schools
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_interventionism
						}
					}
					activate_law = law_type:law_interventionism
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_regulatory_bodies
						}
					}
					activate_law = law_type:law_regulatory_bodies
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_national_guard
						}
					}
					activate_law = law_type:law_national_guard
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_right_of_assembly
						}
					}
					activate_law = law_type:law_right_of_assembly
				}
				if = {
					limit = {
						NOT = {
							has_law = law_type:law_closed_borders
						}
					}
					activate_law = law_type:law_closed_borders
				}
				add_modifier = {
					name = modifier_paris_commune_politics
					months = normal_modifier_time
					is_decaying = yes
				}
				add_modifier = {
					name = modifier_paris_commune_not_dying_economy
					months = normal_modifier_time
				}
				call_election = {
					months = 1
				}
			}
		}
		c:PRC = {
			save_scope_as = paris_commune_scope
			ig:ig_armed_forces = {
				save_scope_as = national_guard_scope
			}
			create_truce = {
				country = c:FRA
				months = 24
			}
		}
		random_state = {
			limit = {
				owner = {
					this = c:PRC
				}
			}
			save_scope_as = paris_scope
		}
		if = {
			limit = {
				socialist_revolution_ongoing = yes
			}
			c:PRC = {
				activate_law = law_type:law_council_republic
			}
		}
	}

	option = {
		name = paris_commune.1.a
		default_option = yes
		add_journal_entry = {
			type = je_the_paris_commune_france
		}
		c:PRC = {
			hidden_effect = {
				add_journal_entry = {
					type = je_the_paris_commune_communards
				}

					value = 0.9
				}

					add_treasury = -100000
				}

		}

paris_commune.2 = {
	type = country_event
	placement = root
	event_image = {
		video = "votp_cops_march"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	title = paris_commune.2.t
	desc = paris_commune.2.d
	flavor = paris_commune.2.f

	duration = 3

	dlc = dlc004

	trigger = {
		
	}

	immediate = {
		set_variable = paris_commune_crushed_var
		c:PRC = {
			save_scope_as = paris_commune_scope
		}
		random_state = {
			limit = {
				owner = {
					this = c:PRC
				}
			}
			save_scope_as = paris_scope
		}
		random_scope_general = {
			limit = {
				any_combat_units = {
					building.state.region = {	
						this = sr:region_france
					}
				}
			}
			save_scope_as = commune_destroyer_general
		}
	}

	option = {
		name = paris_commune.2.a
		default_option = yes
		c:FRA = {
			annex_as_civil_war = c:PRC
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_ILE_DE_FRANCE
			}
			add_radicals_in_state = {
				value = medium_radicals
				pop_type = laborers
			}
			kill_population_percent_in_state = {
				percent = 0.02
				pop_type = soldiers
			}
			kill_population_percent_in_state = {
				percent = 0.02
				pop_type = academics
			}
			kill_population_percent_in_state = {
				percent = 0.02
				pop_type = laborers
			}
			state_region = {
				add_devastation = 15 
			}
		}
		add_modifier = {
			name = modifier_repression_of_commune
			months = long_modifier_time
			is_decaying = yes
		}
		if = {
			limit = {
				any_character_in_void = {
					has_variable = imprisoned_general_var
				}
			}
			random_list = {
				1 = {
					random_character_in_void = {
						limit = {
							has_variable = imprisoned_general_var
						}
						free_character_from_void = yes
					}
				}
				4 = {
					random_character_in_void = {
						limit = {
							has_variable = imprisoned_general_var
						}
						kill_character = yes
					}
				}
			}
		}
	}
	option = {
		name = paris_commune.2.b
		c:FRA = {
			annex_as_civil_war = c:PRC
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_ILE_DE_FRANCE
			}
			kill_population_percent_in_state = {
				percent = 0.01
				pop_type = soldiers
			}
			kill_population_percent_in_state = {
				percent = 0.01
				pop_type = academics
			}
			kill_population_percent_in_state = {
				percent = 0.01
				pop_type = laborers
			}
			state_region = {
				add_devastation = 5 
			}
		}
		if = {
			limit = {
				any_character_in_void = {
					has_variable = imprisoned_general_var
				}
			}
			random_list = {
				1 = {
					random_character_in_void = {
						limit = {
							has_variable = imprisoned_general_var
						}
						free_character_from_void = yes
					}
				}
				2 = {
					random_character_in_void = {
						limit = {
							has_variable = imprisoned_general_var
						}
						kill_character = yes
					}
				}
			}
		}
	}
}
