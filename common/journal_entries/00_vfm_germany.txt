je_trans_german_railway = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	
	group = je_group_tsr_content

	is_shown_when_inactive = {
		NOT = { has_global_variable = trans_german_done }
		OR = {
			country_has_primary_culture = cu:north_german
			country_has_primary_culture = cu:south_german
			country_has_primary_culture = cu:rhenish
		}
		game_date < 1856.1.1
		owns_entire_state_region = STATE_PFALZ
	}

	possible = {
		has_technology_researched = railways
	}

	complete = {
		any_scope_state = {
			state_region = s:STATE_PFALZ
			has_building = building_railway
		}
		any_scope_state = {
			state_region = s:STATE_ELSASS
			has_building = building_railway
		}
		any_scope_state = {
			state_region = s:STATE_WURTTEMBERG
			has_building = building_railway
		}
		any_scope_state = {
			state_region = s:STATE_BAVARIA
			has_building = building_railway
		}
		any_scope_state = {
			state_region = s:STATE_OBERPFALZ
			has_building = building_railway
		}
	}
	
	fail = {
		game_date > 1856.1.1
	}

	on_complete = {
		set_global_variable = trans_german_done
		show_as_tooltip = {
			add_modifier = {
				name = generic_production_tech_research
				months = long_modifier_time
			}
		}
		trigger_event = { id = journal_flavor_germany.1 }
	}

	weight = 50
}
