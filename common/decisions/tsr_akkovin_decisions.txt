centralise_kurdish_territories = {
	is_shown = {
		exists = c:AKN
		root = c:AKN
		has_journal_entry = je_akkovin_reform_kurdish
	}

	possible = {
		is_at_war = no
		custom_tooltip = {
			text = centralise_kurdish_territories_tt
			NOT = { has_variable = centralise_kurdish_vilayets_cooldown }
		}
		any_scope_state = {
			has_modifier = kurdish_autonomy
		}
	}

	when_taken = {
		trigger_event = { id = akkovin_flavor.3 days = 0 popup = yes }
		set_variable = {
			name = centralise_kurdish_vilayets_cooldown
			years = 2
		}
	}

	ai_chance = {
		base = 1000
	}
}

cancel_treaty_of_baglar = {
	is_shown = {
		exists = c:AKN
		root = c:AKN
		has_diplomatic_pact = {
			who = c:PER
			type = treaty_of_baglar_pact
		}
	}

	possible = {
		OR = {
			has_modifier = treaty_of_baglar
			has_modifier = treaty_of_baglar_2
		}
	}

	when_taken = {
		remove_diplomatic_pact = {
			country = c:PER
			type = treaty_of_baglar_pact
		}
		if = {
			limit = { has_modifier = treaty_of_baglar }
			remove_modifier = treaty_of_baglar
		}
		if = {
			limit = { has_modifier = treaty_of_baglar_2 }
			remove_modifier = treaty_of_baglar_2
		}
		change_relations = {
			country = c:PER
			value = -100
		}
		if = {
			limit = {
				any_country = {
					has_diplomatic_pact = { 
						who = root
						type = tob_embargo
					}
				}
			}
			every_country = {
				limit = {
					has_diplomatic_pact = { 
						who = root
						type = tob_embargo
					}
				}
				remove_diplomatic_pact = {
					country = root
					type = tob_embargo
				}
			}
		}
		if = {
			limit = {
				c:PER = {
					has_diplomatic_pact = { 
						who = root
						type = customs_union
					}
				}
			}
			remove_diplomatic_pact = {
				country = c:PER
				type = customs_union
			}
		}
		
		custom_tooltip = no_renewing_baglar_tt
		
		c:PER = { trigger_event = { id = treaty_of_baglar_events.1 days = 1 } }
		
		hidden_effect = {
			every_country = {
				limit = {
					OR = {
						has_interest_marker_in_region = region_arabic
						has_interest_marker_in_region = region_anatolia
						has_interest_marker_in_region = region_kurdistan
						has_interest_marker_in_region = region_persia
					}
				}
				post_notification = akkovin_dont_renew_treaty
			}
		}
	}

	ai_chance = {
		base = 0
	}
}
