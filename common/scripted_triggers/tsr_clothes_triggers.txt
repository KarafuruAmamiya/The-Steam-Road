tsr_persia_leader_trigger = {
	exists = scope:character
	scope:character = {
		exists = home_country
		home_country = {
			NOT = { c:ZNI ?= this }
			OR = {
				country_has_primary_culture = cu:persian
				country_has_primary_culture = cu:jonubi
				AND = {
					exists = c:PER
					is_subject_of = c:PER
					has_variable = adopt_western_uniforms
				}
			}
		}
	}
}

tsr_fez_trigger = {
	exists = scope:character
	scope:character = {
		OR = {
			culture = cu:maghrebi
			culture = cu:akturk
			culture = cu:turkish
			AND = {
				exists = home_country
				home_country = {
					OR = {
						AND = {
							exists = c:TUN
							OR = {
								c:TUN = this
								AND = {
									is_subject_of = c:TUN
									country_is_in_middle_east = yes
								}
							}
						}

							exists = c:MOR
							OR = {

british_empire_clothes_trigger = {
	exists = scope:character
	scope:character = {
		OR = {
			AND = {
				exists = c:GBR
				exists = home_country
				home_country = {
					OR = {
						AND = {
							c:GBR = this
							is_subject = no # preventing uniform conflicts if GBR becomes a subject of another country with puppet uniforms
						}
						is_subject_of = c:GBR
					}		
				}
			}
			AND = {
				exists = c:ENG
				exists = home_country
				home_country = {
					OR = {
						AND = {

							is_subject = no # preventing uniform conflicts if GBR becomes a subject of another country with puppet uniforms
						}
