fra_expand_japan = {
	is_shown = {
		country_is_in_europe = yes
		exists = c:UES
		c:UES = {
			is_subject_of = ROOT
			is_subject_type = subject_type_dominion
		}
	}
	possible = {
		OR = { 
			any_state = {
				owner = ROOT
				region = sr:region_japan
			}
			any_subject_or_below = {
				capital.region = sr:region_japan
				NOT = { 
					THIS = c:UES
					is_subject_of = c:UES
				}
			}
		}
	}
	when_taken = {
		custom_tooltip = expand_japan_tt
		if = {
			limit= {
				any_state = {
					owner = ROOT
					region = sr:region_japan
				}
			}
			every_scope_state = {
				limit = {
					owner = ROOT
					region = sr:region_japan
				}
				set_state_owner = c:UES
				set_state_type = unincorporated
			}
		}
		if = {
			limit= {
				any_subject_or_below = {
					capital.region = sr:region_japan
					NOT = { 
						THIS = c:UES 
						is_subject_of = c:UES 
					}
				}
			}
			every_subject_or_below = {
				if = {
					limit = { 
						is_subject_type = subject_type_protectorate
						capital.region = sr:region_japan
						NOT = { 
							THIS = c:UES 
							is_subject_of = c:UES 
						} 						
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_ues_protectorate
					c:UES = {
						create_diplomatic_pact = {
							country = scope:new_ues_protectorate
							type = tributary
						}
					}
				}
				if = {
					limit = { 
						is_subject_type = subject_type_vassal
						capital.region = sr:region_japan
						NOT = { 
							THIS = c:UES 
							is_subject_of = c:UES 
						} 
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_ues_vassal
					c:UES = {
						create_diplomatic_pact = {
							country = scope:new_ues_vassal
							type = vassal
						}
					}
				}
				if = {
					limit = { 
						is_subject_type = subject_type_puppet
						capital.region = sr:region_japan
						NOT = { 
							THIS = c:UES 
							is_subject_of = c:UES 
						} 
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_ues_puppet
					c:UES = {
						create_diplomatic_pact = {
							country = scope:new_ues_puppet
							type = vassal
						}
					}
				}
			}
		}
	}
	ai_chance = {
		base = 100
	}
}
