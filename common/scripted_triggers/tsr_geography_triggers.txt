### State geography triggers

country_is_in_new_word = {
	exists = capital
	capital = {
		OR = {
			state_is_in_americas = yes
			state_is_in_oceania = yes
		}
	}
}

country_is_in_hre = {
	exists = capital
	capital = {
		OR = {
			region = sr:region_north_germany
			region = sr:region_south_germany
			region = sr:region_rhine
			region = sr:region_low_countries
			state_region = s:STATE_UPPER_SILESIA
			state_region = s:STATE_WEST_SLOVAKIA
		}
	}
}

gunpowder_empire_trigger = {
	OR = {
		c:AKN ?= this
		c:PER ?= this
		c:TUR ?= this
	}
}
