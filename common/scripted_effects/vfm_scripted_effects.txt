set_character_as_prime_minister = {
	custom_tooltip = {
		text = set_character_as_prime_minister
		owner = {
			random_scope_character = {
				limit = {
					has_variable = prime_minister_trigger
				}
				remove_variable = prime_minister_trigger
			}
		}
		set_variable  = prime_minister_trigger
		owner = {
			set_variable = {
				name = prime_minister
				value = PREV
			}
		}
	}
}
