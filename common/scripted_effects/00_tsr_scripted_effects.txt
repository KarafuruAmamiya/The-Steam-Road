do_odrodzenie_math = {
	set_variable = { name = polish_states_pop_total value = 0 }
	set_variable = { name = polish_states_radicals_total value = 0 }
	every_scope_state = {
		limit = {
			state_region = {
				is_homeland = cu:polish
			}
		}
		set_variable = { name = polish_state_pop_radicals value = this.turmoil }
		change_variable = { name = polish_state_pop_radicals multiply = this.state_population }
		ROOT = {
			change_variable = { name = polish_states_pop_total add = prev.state_population }
			change_variable = { name = polish_states_radicals_total add = prev.var:polish_state_pop_radicals }
		}
	}
	set_variable = { name = polish_states_radicals_total_bar value = var:polish_states_radicals_total }
	change_variable = { name = polish_states_radicals_total_bar divide = var:polish_states_pop_total }
	change_variable = { name = polish_states_radicals_total_bar multiply = 100 }
}

calculate_russian_unrest = {
	set_variable = { name = russian_unrest_var_next value = 0 }
	set_variable = { name = russian_unrest_from_radicals value = 0 }
	every_scope_state = {
		limit = {
			state_region = {
				is_homeland = cu:russian
			}
			turmoil >= 0.10
		}
		ROOT = {
			change_variable = { name = russian_unrest_from_radicals add = prev.turmoil }
		}
	}
	set_variable = { name = russian_unrest_from_loyalists value = 0 }
	every_scope_state = {
		limit = {
			state_region = {
				is_homeland = cu:russian
			}
			loyalty >= 0.10
		}
		ROOT = {
			change_variable = { name = russian_unrest_from_loyalists add = prev.loyalty }
		}
	}
	change_variable = { name = russian_unrest_from_loyalists multiply = -1 }
	#From Literacy
	set_variable = { name = russian_unrest_from_literacy value = 0 }
	every_scope_state = {
		limit = {
			state_region = {
				is_homeland = cu:russian
			}
		}
		set_variable = { name = temp_literacy value = this.literacy_rate }
		change_variable = { name = temp_literacy multiply = this.state_population }
		change_variable = { name = temp_literacy divide = root.total_population }
		change_variable = { name = temp_literacy multiply = 10 } # This variable equals the percentage of pops in incorporated states below expected SoL times ten. If 25% of your pops in incorporated states are below expected SoL, this will equal 2.5.
		ROOT = {
			change_variable = { name = russian_unrest_from_literacy add = prev.var:temp_literacy }
		}
	}
	#From Nationalism
	set_variable = { name = russian_unrest_from_nationalism value = 0 }
	if = {
		limit = {
			has_technology_researched = nationalism
		}
		ROOT = {
			change_variable = { name = russian_unrest_from_nationalism add = 1 }
		}
	}
	#From Pan-Nationalism
	set_variable = { name = russian_unrest_from_pannationalism value = 0 }
	if = {
		limit = {
			has_technology_researched = pan-nationalism
		}
		ROOT = {
			change_variable = { name = russian_unrest_from_pannationalism add = 1 }
		}
	}
	#From Academia
	set_variable = { name = russian_unrest_from_universities value = 0 }
	every_scope_building = {
		limit = {
			OR = {
				is_building_type = building_arts_academy
				is_building_type = building_university
			}
			occupancy >= 0.75
		}
		ROOT = {
			change_variable = { name = russian_unrest_from_universities add = prev.level }
		}
	}
	change_variable = { name = russian_unrest_from_universities divide = 25 }
	# From Laws
	set_variable = { name = russian_unrest_from_laws value = 0 }
	# Schools
	set_variable = { name = temp_schooling_var_contribution value = institution:institution_schools.investment }
	if = {
		limit = {
			has_law = law_type:law_no_schools
		}
		change_variable = { name = temp_schooling_var_contribution add = 0.3 }
	}
	else_if = {
		limit = {
			has_law = law_type:law_private_schools
		}
		change_variable = { name = temp_schooling_var_contribution multiply = -0.1 }
	}
	else_if = {
		limit = {
			has_law = law_type:law_religious_schools
		}
		change_variable = { name = temp_schooling_var_contribution multiply = -0.3 }
	}
	else_if = {
		limit = {
			has_law = law_type:law_public_schools
		}
		change_variable = { name = temp_schooling_var_contribution multiply = -0.5 }
	}
	# Cops
	set_variable = { name = temp_policing_var_contribution value = institution:institution_police.investment }
	if = {
		limit = {
			has_law = law_type:law_no_police
		}
		change_variable = { name = temp_policing_var_contribution add = 0.3 }
	}
	else_if = {
		limit = {
			has_law = law_type:law_local_police
		}
		change_variable = { name = temp_policing_var_contribution multiply = -0.1 }
	}
	else_if = {
		limit = {
			has_law = law_type:law_dedicated_police
		}
		change_variable = { name = temp_policing_var_contribution multiply = -0.3 }
	}
	else_if = {
		limit = {
			has_law = law_type:law_militarized_police
		}
		change_variable = { name = temp_policing_var_contribution multiply = -0.5 }
	}
	# Internal Affairs
	set_variable = { name = temp_ha_var_contribution value = institution:institution_home_affairs.investment }
	if = {
		limit = {
			has_law = law_type:law_no_home_affairs
		}
		change_variable = { name = temp_ha_var_contribution add = 0.3 }
	}
	else_if = {
		limit = {
			has_law = law_type:law_guaranteed_liberties
		}
		change_variable = { name = temp_ha_var_contribution multiply = -0.1 }
	}
	else_if = {
		limit = {
			has_law = law_type:law_national_guard
		}
		change_variable = { name = temp_ha_var_contribution multiply = -0.3 }
	}
	else_if = {
		limit = {
			has_law = law_type:law_secret_police
		}
		change_variable = { name = temp_ha_var_contribution multiply = -0.5 }
	}
	change_variable = { name = russian_unrest_from_laws add = var:temp_schooling_var_contribution }
	change_variable = { name = russian_unrest_from_laws add = var:temp_policing_var_contribution }
	change_variable = { name = russian_unrest_from_laws add = var:temp_ha_var_contribution }
	remove_variable = temp_schooling_var_contribution
	remove_variable = temp_policing_var_contribution
	remove_variable = temp_ha_var_contribution

	# Compilation
	change_variable = { name = russian_unrest_var_next add = var:russian_unrest_from_radicals }
	change_variable = { name = russian_unrest_var_next add = var:russian_unrest_from_loyalists }
	change_variable = { name = russian_unrest_var_next add = var:russian_unrest_from_literacy }
	change_variable = { name = russian_unrest_var_next add = var:russian_unrest_from_nationalism }
	change_variable = { name = russian_unrest_var_next add = var:russian_unrest_from_pannationalism }
	change_variable = { name = russian_unrest_var_next add = var:russian_unrest_from_universities }
	change_variable = { name = russian_unrest_var_next add = var:russian_unrest_from_laws }
	change_variable = { name = russian_unrest_var_next add = var:russian_unrest_var_add }
}
