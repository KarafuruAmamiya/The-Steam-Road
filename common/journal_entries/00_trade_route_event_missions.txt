je_build_local_arms_industry = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	complete = {
		any_scope_state = {
			has_active_building = building_arms_industry
		}
	}

	on_complete = {
		ig:ig_armed_forces = {
			add_modifier = {
				name = ig_domestic_weapons_manufacturing
				months = short_modifier_time
			}
		}
	}


	on_timeout = {
		ig:ig_armed_forces = {
			add_modifier = {
				name = ig_unreliable_weapon_imports
				months = short_modifier_time
			}
		}
	}

	timeout = 600

	weight = 200

	transferable = no
}

je_set_up_grain_import = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	complete = {
		capital.market = {
			any_trade_route = {
				importer.owner = root
				goods = g:grain
			}
		}
	}

	on_complete = {
		scope:target = {
			add_modifier = {
				name = ig_secured_desired_imports
				months = normal_modifier_time
			}
		}
	}


	on_timeout = {
		scope:target = {
			add_modifier = {
				name = ig_refused_desired_imports
				months = normal_modifier_time
			}
		}
	}

	timeout = 600

	weight = 200

	transferable = no
}

je_set_up_tools_import = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"
	
	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_ISFAHAN
			}
			save_scope_as = isfahan_scope
		}
	}

	complete = {
		capital.market = {
			mg:tools = { market_goods_cheaper <= 0.05 }
		}
	}

	on_complete = {
		ig:ig_trade_unions = {
			add_modifier = {
				name = ig_secured_desired_imports
				months = normal_modifier_time
			}
		}
		ig:ig_petty_bourgeoisie = {
			add_modifier = {
				name = ig_secured_desired_imports
				months = normal_modifier_time
			}
		}
	}


	on_timeout = {
		ig:ig_trade_unions = {
			add_modifier = {
				name = ig_refused_desired_imports
				months = normal_modifier_time
			}
		}
		ig:ig_petty_bourgeoisie = {
			add_modifier = {
				name = ig_refused_desired_imports
				months = normal_modifier_time
			}
		}
	}

	timeout = 1825

	weight = 200

	transferable = no
}
