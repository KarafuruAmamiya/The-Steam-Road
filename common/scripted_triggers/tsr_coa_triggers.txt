coa_non_colonial_overlord_trigger = {
	exists = scope:target
	scope:target = { 
		is_non_customs_union_subject = yes
		overlord = {
			NOT = { is_country_type = colonial }
		}
	}
}

coa_def_scandinavian_ensign_trigger = {
	exists = c:SWE
	c:SWE = { save_temporary_scope_as = custom_overlord }
	coa_def_ensign_trigger = yes
}

coa_def_persian_ensign_trigger = {
	exists = c:PER
	c:PER = { save_temporary_scope_as = custom_overlord }
	coa_def_ensign_trigger = yes
}
