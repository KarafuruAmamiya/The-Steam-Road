# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

law_feudalism = {
	group = lawgroup_centralization
	
	icon = "gfx/interface/icons/law_icons/peasant_levies.dds"
	
	progressiveness = -100
	
	possible_political_movements = {
		law_unitary
		law_federal
	}

	disallowing_laws = {
		law_anarchy
	}
	
	# AI should never switch back to feudalism
	ai_will_do = {
		always = no
	}

	modifier = {
		country_authority_add = -50
		country_aristocrats_pol_str_mult = 0.25
		state_tax_capacity_mult = -0.2
		state_bureaucracy_population_base_cost_factor_mult = -0.25
		state_radicals_from_sol_change_mult = 0.25
	}
	
	pop_support = {
		value = 0
		add = {
			desc = "POP_ARISTOCRATS"			
			if = {
				limit = { 
					is_pop_type = aristocrats
				}
				value = 0.33
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}

law_warlord_state = {
	group = lawgroup_centralization
	
	icon = "gfx/interface/icons/law_icons/unused/revolutionary_conscription.dds"
	
	can_enact = {
		NOT = { country_tier = empire }
	}
	
	possible_political_movements = {
		law_unitary
		law_federal
	}
	
	# AI should never become warlord voluntarily
	ai_will_do = {
		always = no
	}
	
	progressiveness = -50
	
	modifier = {
		country_authority_add = -50
		country_officers_pol_str_mult = 0.5
		state_tax_capacity_mult = -0.2
		state_mortality_mult = 0.025
		state_bureaucracy_population_base_cost_factor_mult = -0.25
		country_radicals_from_conquest_mult = -0.25
		state_radicals_from_sol_change_mult = 0.25
	}
	
	on_activate = {
		if = {
			limit = {
				NOT = { has_law = law_type:law_serfdom }
				ig:ig_landowners = {
					has_ideology = ideology:ideology_stratocratic
				}
			}
			ig:ig_landowners = {
				remove_ideology = ideology_stratocratic
				add_ideology = ideology_modern_stratocratic
			}
		}
	}
	
	pop_support = {
		value = 0
		add = {
			desc = "POP_OFFICERS"			
			if = {
				limit = { 
					is_pop_type = officers
				}
				value = 0.33
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}

		state_radicals_from_discrimination_mult = -0.25
		country_bureaucracy_investment_cost_factor_mult = -0.5
	}
	
	possible_political_movements = {
		law_devolved
		law_unitary
		law_federal
	}
	
	on_activate = {
		custom_tooltip = cant_annex_subject_tt
		if = {
			limit = {
				NOT = { is_country_type = decentralized }
				NOT = { has_law = law_type:law_serfdom }
				ig:ig_landowners = {
					has_ideology = ideology:ideology_stratocratic
				}
			}
			ig:ig_landowners = {
				remove_ideology = ideology_stratocratic
				add_ideology = ideology_modern_stratocratic
			}
		}
	}

	pop_support = {
		value = 0
		add = {
			desc = "POP_BUREAUCRATS"
			if = {
				limit = {
					is_pop_type = bureaucrats
				}
			}
			value = 0.25
		}
		add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					culture_accepted = no
				}
				add = 0.1
				if = {
					limit = { 
						standard_of_living <= 10
					}
					add = 0.1
				}
				multiply = { #We want independance instead!
					value = literacy_rate
					add = -1
					multiply = -1
				}		
			}		
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
