je_swe_regency = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	group = je_group_tsr_content

	complete = {
		trigger_if = {
			limit = {
				any_scope_character = {
					has_variable = is_gustav_iii
				}
			}
			custom_tooltip = {
				text = gustav_come_to_age_tt
				trigger_if = {
					limit = { has_variable = faster_regency }
					any_scope_character = {
						has_variable = is_gustav_iii
						age >= 14
					}
				}
				trigger_else = {
					any_scope_character = {
						has_variable = is_gustav_iii
						age >= 16
					}
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = heir_come_to_age_tt
				any_scope_character = {
					is_heir = yes
					age >= 16
				}
			}
		}
	}

	on_complete = {
		trigger_event = { id = scandinavian_events.1 }
		hidden_effect = {
			c:DEN = { trigger_event = { id = scandinavian_events.1 } }
			c:NOR = { trigger_event = { id = scandinavian_events.1 } }
		}
	}
	
	on_monthly_pulse = {
		events = {
			scandinavian_events.2
		}
		random_events = {
			100 = 0
			10 = scandinavian_events.3
		}
	}
	
	fail = {
		custom_tooltip = {
			text = gustav_iii_coronated_tt
			ruler = { NOT = { has_variable = is_eleonora } }
		}
	}

	weight = 10

	should_be_pinned_by_default = yes
}
