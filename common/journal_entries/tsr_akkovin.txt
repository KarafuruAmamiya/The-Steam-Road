je_akkovin_crisis = { # Crisis of the Akkovin Confederation
	icon = "gfx/interface/icons/event_icons/event_revolt.dds"
	
	should_be_pinned_by_default = yes

	complete = {
		c:SYR = {
			is_subject_of = ROOT
			is_subject_type = subject_type_dependency
			has_attitude = {
				who = ROOT
				attitude = loyal
			}
			NOT = { has_variable = arab_uprising }
		}
		c:HDJ = {
			is_subject_of = ROOT
			is_subject_type = subject_type_dependency
			has_attitude = {
				who = ROOT
				attitude = loyal
			}
			NOT = { has_variable = arab_uprising }
		}
		c:NEJ = {
			is_subject_of = ROOT
			is_subject_type = subject_type_dependency
			has_attitude = {
				who = ROOT
				attitude = loyal
			}
			NOT = { has_variable = arab_uprising }
		}
		c:JAB = {
			is_subject_of = ROOT
			is_subject_type = subject_type_dependency
			has_attitude = {
				who = ROOT
				attitude = loyal
			}
			NOT = { has_variable = arab_uprising }
		}
		every_scope_state = {	
			turmoil < 0.20
		}
		has_variable = akc_reforms_complete
	}
	
	fail = {
		OR = {
			any_scope_state = {
				limit = {
					is_incorporated = yes
				}
				count < 9
			}
			NOT = { c:SYR = { is_subject_of = ROOT } }
			NOT = { c:JAB = { is_subject_of = ROOT } }
			NOT = { c:HDJ = { is_subject_of = ROOT } }
			NOT = { c:NEJ = { is_subject_of = ROOT } }
			exists = c:ARA
		}
	}
	
	on_fail = {
		trigger_event = { id = akkovin_crisis.2 }
	}

	on_complete = {
		trigger_event = { id = akkovin_crisis.3 }
	}
	
	weight = 10000
	
	inheritable = no

	invalid = {
		NOT = { exists = c:AKN }
	}
}

	on_fail = {

	}

	current_value = {

	}

	goal_add_value = {

	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
	
	invalid = {
		NOT = { exists = c:AKN }
	}
}
