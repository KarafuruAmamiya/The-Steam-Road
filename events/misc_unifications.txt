# AMR, USC
formation.6 = {
	type = country_event
	placement = ROOT

	title = formation.6.t
	desc = formation.6.d
	flavor = {
		first_valid = {
			triggered_desc = {
				desc = formation.6.f
				trigger = {
					c:AMR ?= THIS
				}
			}
			triggered_desc = {
				desc = formation.6.f2
				trigger = {
					c:USC ?= THIS
				}
			}
		}
	}

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
		OR = {
			AND = {
				c:AMR ?= THIS
				NOT = { has_global_variable = has_formed_brz }
			}
			AND = {
				c:USC ?= THIS
				NOT = { has_global_variable = has_formed_usa }
			}
		}
	}

	immediate = {
		if = {
			limit = {
				c:AMR ?= THIS
			}
			set_global_variable = has_formed_brz
		}
		else_if = {
			limit = {
				c:USC ?= THIS
			}
			set_global_variable = has_formed_usa
		}
		save_scope_as = formation_country
		every_country = {
			limit = {
				has_diplomatic_relevance = ROOT
			}
			post_notification = reunification_notification
		}
	}

	option = {
		name = formation.5.a
		default_option = yes
		add_modifier = {
			name = unification_prestige
			months = very_long_modifier_time
		}
		unification_claims_effect = yes
		ai_chance = {
			factor = 1
		}
	}
}
